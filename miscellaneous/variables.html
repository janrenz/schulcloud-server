apps/server/src/core/error/filter/global-error.filter.ts

createErrorResponse
Default value : (error: unknown, logger: ILogger): ErrorResponse => { try { if (error instanceof Error) { if (isFeathersError(error)) { // handles feathers errors only when calling feathers services from nest app return createErrorResponseForFeathersError(error); } if (isBusinessError(error)) { // create response from business error using 409/conflict return createErrorResponseForBusinessError(error); } if (isTechnicalError(error)) { // create response from technical error return createErrorResponseForHttpException(error); } } // create default response for all unknown errors return createErrorResponseForUnknownError(); } catch (exception) { const stack = exception instanceof Error ? exception.stack : undefined; logger.error(exception, stack, 'Response Error'); return createErrorResponseForUnknownError(); } }
createErrorResponseForHttpException
Default value : (exception: HttpException): ErrorResponse => { const code = exception.getStatus(); const msg = exception.message || 'Some error occurred'; const exceptionName = exception.constructor.name.replace('Exception', ''); const type = _.snakeCase(exceptionName).toUpperCase(); const title = _.startCase(exceptionName); return new ErrorResponse(type, title, msg, code); }

Creates ErrorResponse from NestJS build in technical exceptions
isBusinessError
Default value : (error: Error): error is BusinessError => { return error instanceof BusinessError; }
isFeathersError
Default value : (error: Error): error is FeathersError => { if (!(error && 'type' in error)) return false; return (error as FeathersError)?.type === 'FeathersError'; }
isTechnicalError
Default value : (error: Error): error is HttpException => { return error instanceof HttpException; }

Compare helper to detect an error is a build in NestJS http exception.
writeErrorLog
Default value : (error: unknown, logger: ILogger): void => { if (error instanceof Error) { if (isFeathersError(error)) { logger.error(error, error.stack, 'Feathers Error'); } else if (isBusinessError(error)) { if (error instanceof ApiValidationError) { writeValidationErrors(error, logger); } else { logger.error(error, error.stack, 'Business Error'); } } else if (isTechnicalError(error)) { logger.error(error, error.stack, 'Technical Error'); } else { logger.error(error, error.stack, 'Unhandled Error'); } } else { logger.error(error, 'Unknown error'); } }
writeValidationErrors
Default value : (error: ApiValidationError, logger: ILogger) => { const errorMsg = error.validationErrors.map( // eslint-disable-next-line @typescript-eslint/restrict-template-expressions (e) => `Wrong property ${e.property} got ${e.value} : ${JSON.stringify(e.constraints)}` ); logger.error(errorMsg, error.stack, 'API Validation Error'); }
